CREATE TABLE stg_customer (
    customer_id INT,
    name VARCHAR(100),
    address VARCHAR(200),
    phone VARCHAR(20),
    last_updated DATETIME2(0)
);

CREATE TABLE dim_customer (
    customer_id INT,
    name VARCHAR(100),
    address VARCHAR(200),
    phone VARCHAR(20),
    start_date DATETIME2(0),
    end_date DATETIME2(0) NULL,
    is_current INT
);


CREATE OR ALTER PROCEDURE sp_upsert_customer_scd2
AS
BEGIN

    -- Step 1 + 2: Close old versions if data changed
    WITH latest_customer AS (
        SELECT *
        FROM (
            SELECT *,
                   ROW_NUMBER() OVER (
                       PARTITION BY customer_id
                       ORDER BY last_updated DESC
                   ) AS rn
            FROM stg_customer
        ) ranked
        WHERE rn = 1
    )

    UPDATE dc
    SET dc.end_date = lc.last_updated,
        dc.is_current = 0
    FROM dim_customer dc
    JOIN latest_customer lc
      ON dc.customer_id = lc.customer_id
     AND dc.end_date IS NULL
     AND (
            dc.name <> lc.name
         OR dc.address <> lc.address
         OR dc.phone <> lc.phone
     );



    -- Step 3: Insert new customers or changed versions
    WITH latest_customer AS (
        SELECT *
        FROM (
            SELECT *,
                   ROW_NUMBER() OVER (
                       PARTITION BY customer_id
                       ORDER BY last_updated DESC
                   ) AS rn
            FROM stg_customer
        ) ranked
        WHERE rn = 1
    )

    INSERT INTO dim_customer (
        customer_id,
        name,
        address,
        phone,
        start_date,
        end_date,
        is_current
    )
    SELECT
        lc.customer_id,
        lc.name,
        lc.address,
        lc.phone,
        lc.last_updated,
        NULL,
        1
    FROM latest_customer lc
    LEFT JOIN dim_customer dc
      ON lc.customer_id = dc.customer_id
     AND dc.end_date IS NULL
    WHERE dc.customer_id IS NULL
       OR dc.name <> lc.name
       OR dc.address <> lc.address
       OR dc.phone <> lc.phone;

END;
